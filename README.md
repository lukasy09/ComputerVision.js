<header><h1>ComputerVision.js</h1></header>

        <li>addition</li>
        <li>subtraction</li>
        <li>linear combination</li>
        <li>Convolution</li>
        <li>Gaussian filtering</li>
        <li>Laplacian filtering</li>
    </ol>
